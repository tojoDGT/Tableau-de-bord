SELECT SQL_CALC_FOUND_ROWS *,entite_denomination AS Institution,
typeEntite_libelle AS Secteur,
produit_trimestreId AS trimestre,produit_annee AS Année,
region_libelle AS Region,
informationAssurance_totalEffectif,informationAssurance_pourcentage,'',''

FROM `cnfi_produit` 
INNER JOIN `cnfi_entite` ON produit_entiteId = entite_id
INNER JOIN `cnfi_typeentite` ON   typeEntite_id = entite_typeEntiteId
INNER JOIN `cnfi_zone` ON zone_produitId = produit_id
INNER JOIN `cnfi_informationassurance` ON produitId = produit_id 
LEFT JOIN `cnfi_region` ON region_id = zone_regionId
WHERE 1 %WHERE%
WHERE informationAssurance_totalEffectif >0 OR informationAssurance_pourcentage > 0
GROUP BY produit_id
ORDER BY produit_trimestreId ASC,produit_annee ASC,entite_denomination ASC
